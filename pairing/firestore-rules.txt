rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // קודי מדריכה – קריאה בלבד (בדיקת קוד)
    match /instructors/{code} {
      allow read: if true;
      allow write: if false;
    }

    // סשן כיתתי – נוצר ומעודכן רק עם קוד מדריכה תקף
    match /sessions/{sessionId} {
      allow create, update: if
        request.resource.data.instructorCode is string &&
        exists(
          /databases/$(database)/documents/instructors/$(request.resource.data.instructorCode)
        );

      allow read: if true;
      allow delete: if false;
    }

    // תשובות תלמידות – יצירה בלבד, רק אם קיים סשן
    match /responses/{responseId} {
      allow create: if
        request.resource.data.sessionId is string &&
        exists(
          /databases/$(database)/documents/sessions/$(request.resource.data.sessionId)
        );

      allow read: if true;
      allow update, delete: if false;
    }

    // חסימה מוחלטת לכל דבר אחר
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
